dom4j生成xml文件 炒菜  发布于 2015-11-12 22:20:33

    在前面一篇，我们介绍了dom4j怎么解析xml文件，这里我们介绍怎么通过dom4j来生成xml文件。

1、导入需要的jar包：dom4j-1.6.1.jar，这个jar包在前面一篇的例子中有下载。

http://javaxuexi.net/jx/webjichu/20151112/1148.html

2、在com.javaxuexi.dom包下创建生成xml文件的类：Create_Xml 

package com.javaxuexi.dom;

import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.OutputStream;

import org.dom4j.Document;
import org.dom4j.DocumentFactory;
import org.dom4j.Element;
import org.dom4j.io.OutputFormat;
import org.dom4j.io.XMLWriter;

public class Create_Xml {

    public static void main(String[] args) {
        // TODO Auto-generated method stub
        DocumentFactory factory=new DocumentFactory();//创建工厂
        Document document=factory.createDocument();//创建文档
        //下面都是创建所有的子节点孙节点信息
        Element books=factory.createElement("books");
        Element book1=factory.createElement("book");
        Element name1=factory.createElement("name");
        Element author1=factory.createElement("author");
        name1.setText("《西游记》");
        author1.setText("吴承恩");
        Element book2=factory.createElement("book");
        Element name2=factory.createElement("name");
        Element author2=factory.createElement("author");
        name2.setText("《三国演义》");
        author2.setText("罗贯中");
        Element book3=factory.createElement("book");
        Element name3=factory.createElement("name");
        Element author3=factory.createElement("author");
        name3.setText("《水浒传》");
        author3.setText("施耐庵");
        Element book4=factory.createElement("book");
        Element name4=factory.createElement("name");
        Element author4=factory.createElement("author");
        name4.setText("《红楼梦》");
        author4.setText("曹雪芹");
        //把孙节点信息加入到子节点下面
        book1.add(name1);
        book1.add(author1);
        book2.add(name2);
        book2.add(author2);
        book3.add(name3);
        book3.add(author3);
        book4.add(name4);
        book4.add(author4);
        //把子节点加入到根节点下面
        books.add(book1);
        books.add(book2);
        books.add(book3);
        books.add(book4);
        //把根节点加入到文档中
        document.add(books);
        File file=new File("src/com/javaxuexi/dom/mingzhu.xml");//要创建的xml文件
        OutputStream os=null;
        XMLWriter writer=null;
        try {
            os=new FileOutputStream(file);
            OutputFormat format=new OutputFormat("  ", true);
            writer=new XMLWriter(os,format);
            writer.write(document);//通过xmlwriter生成xml文件
            writer.flush();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }finally{
            try {
                writer.close();
                os.close();
            } catch (IOException e) {
                // TODO Auto-generated catch block
                e.printStackTrace();
            }
        }
        System.out.println(file.getName()+"文件创建完成");
    }

}


3、运行结果：

刷新工程，我们会看到在com.javaxuexi.dom包下生成了新的xml文件：mingzhu.xml
