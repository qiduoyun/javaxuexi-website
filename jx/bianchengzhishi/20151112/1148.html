dom4j解析xml文件 炒菜  发布于 2015-11-12 21:42:40

    xml的用途非常广泛，常见于配置文件，比如web.xml就是xml文件。xml也作为数据交换的协议。那么怎么解析xml文件呢？

    这里介绍使用dom4j来解析xml文件。

1、首先在工程中导入dom4j-1.6.1.jar，这个jar包在源码中一起提供了。

2、包com.javaxuexi.dom包下创建一个xml文件：book.xml，我们就以解析这个文件为例子。 

<?xml version="1.0" encoding="UTF-8"?>
<books>
  <book>
    <name>《三国演义》</name>
    <author>罗贯中</author>
  </book>
  <book>
    <name>《红楼梦》</name>
    <author>曹雪芹</author>
  </book>
  <book>
    <name>《水浒传》</name>
    <author>施耐庵</author>
  </book>
  <book>
    <name>《西游记》</name>
    <author>吴承恩</author>
  </book>
</books>

3、创建xml文件解析类：Read_Xml 

package com.javaxuexi.dom;

import java.io.File;
import java.util.List;

import org.dom4j.Document;
import org.dom4j.DocumentException;
import org.dom4j.Element;
import org.dom4j.io.SAXReader;

public class Read_Xml {

    public static void main(String[] args) {
        // TODO Auto-generated method stub
        File file=new File("src/com/javaxuexi/dom/book.xml");//xml文件
        try {
            SAXReader reader=new SAXReader();//创建解析对象
            Document document=reader.read(file);//获得xml文件的document
            Element root=document.getRootElement();//获取根节点books
            Listelements=root.elements();//获取根节点下的所有子节点book
            for(Element element:elements){
                Listbook=element.elements();//books下的所有孙节点
                for(Element info:book){//输出孙节点的信息
                    System.out.print(info.getName()+"\t"+info.getText()+"\t");
                }
                System.out.println();//输出完一个book的信息换行
            }
        }catch (DocumentException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
    }

}
